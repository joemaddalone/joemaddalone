Now we can hang some meaningful items on our grid positioned by column and row.

.container {
  display: grid;
  grid-template-columns: 20% auto 20%;
  grid-gap: 15px;
}

.logo {
  grid-column: 1;
  grid-row: 1;
}

nav {
  grid-column: 1;
  grid-row: 2;
}

aside {
  grid-column: 3;
  grid-row: 1;
}

  grid-column: 2;
  grid-row: 1;
